int key1(){
	asm("mov r3, pc\n");
}
int key2(){
	asm(
	"push	{r6}\n"
	"add	r6, pc, $1\n"
	"bx	r6\n"
	".code   16\n"
	"mov	r3, pc\n"
	"add	r3, $0x4\n"
	"push	{r3}\n"
	"pop	{pc}\n"
	".code	32\n"
	"pop	{r6}\n"
	);
}
int key3(){
	asm("mov r3, lr\n");
}
int main(){
	int key=0;
	printf("Daddy has very strong arm! : ");
	scanf("%d", &key);
	if( (key1()+key2()+key3()) == key ){

		int r = read(fd, buf, 100);
		write(0, buf, r);
	}
	else{
		printf("I have strong leg :P\n");
	}
	return 0;
}

It sums up three keys:

# key1()

Key1 basically does:

1

mov r3, pc

In ARM there are two modes. After trial and error, we find out the instance is running in thumb mode. So pc is next_next_instruction = 0x8ce4

# key2()

There are a bunch of code to confuse you, but what we mainly care about is the mov r3, pc and adds r3, $4q. So it is next_next_instruction + 4 = 0x8d08 + 4 = 0x8d0c

# key3()

1

mov	r3, lr

lr stores the return address of the function call. So, 0x8d80

# Done

Sum those all up and you get the key:
