def letter_order(string):
    counts = [0]*26
    order = ""
    for i in string:
        c = ord(i.upper()) - 65
        if -1 < c < 26:
            counts[c] += 1
    for i in range(26):
        m = (0,0)
        for j in range(26):
            if counts[j] > m[1]:
                m = (j,counts[j])
        counts[m[0]] = 0
        order += chr(65+m[0])
    return order

def genkeys(order):
    if len(order) == 0:
        return [""]
    keys = []
    for i in range(len(order[0])):
        if len(order[0]) > 1:
            neworder = [order[0][0:i] + order[0][i+1:]] + order[1:]
        else:
            neworder = order[1:]
        for j in genkeys(neworder):
            keys.append(order[0][i]+j)
    return keys

def solve(s,orig,ceasar):
    solved = ""
    for i in s:
        if 64 < ord(i.upper()) < 91:
            solved += orig[ceasar.find(i.upper())]
    return solved
